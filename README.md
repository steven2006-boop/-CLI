Η επιχείρηση δεν πουλά το λογισμικό, αλλά προσφέρει υπηρεσίες όπως τακτικούς ελέγχους περιεχομένου, παρακολούθηση λειτουργίας, ενημερώσεις και τεχνική υποστήριξη.

Με αυτόν τον τρόπο συνδυάζεται η φιλοσοφία του ανοικτού λογισμικού με ένα βιώσιμο επιχειρηματικό μοντέλο βασισμένο στις υπηρεσίες και όχι στην ιδιοκτησία του κώδικα.

import requests

def get_latest_posts():
    url = "http://localhost/wordpress/wp-json/wp/v2/posts?per_page=3"
    
    try:
        response = requests.get(url)
        response.raise_for_status()
        posts = response.json()

        print("\nΤελευταία 3 Άρθρα WordPress:\n")

        for i, post in enumerate(posts, start=1):
            title = post["title"]["rendered"]
            print(f"{i}. {title}")

    except requests.exceptions.RequestException as e:
        print("Σφάλμα σύνδεσης με το WordPress API:", e)

if __name__ == "__main__":
    get_latest_posts()
